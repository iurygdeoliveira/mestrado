##########
# DOCKER #
##########

- Verificar quanto de espaço está sendo ocupado
sudo docker system df
sudo docker system df -v

- Remover todas as imagens
sudo docker system prune -a

- Remover containers, volumes e imagens
docker stop $(docker ps -aq) && docker rm $(docker ps -aq) && docker volume rm $(docker volume ls -qf dangling=true) && docker rmi $(docker images -q) 

Verificar quais imagens estão presentes na máquina
sudo docker images

Remover uma imagem
sudo docker rmi < id imagem > --force

Remover um arquivo grande a arvore do git
git filter-branch -f --prune-empty -d /dev/shm/scratch \
  --index-filter "git rm --cached -f --ignore-unmatch docker-compose.yml" \
  --tag-name-filter cat -- --all
